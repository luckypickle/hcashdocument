360  inline bool equals(GluaStorageValue &other)
361  {
362  if (type != other.type)
363  return false;
364  switch (type)
365  {
366  case hsrcore::blockchain::StorageValueTypes::storage_value_string:
367  return strcmp(value.string_value, other.value.string_value) == 0;
368  case hsrcore::blockchain::StorageValueTypes::storage_value_int:
369  return value.int_value == other.value.int_value;
370  case hsrcore::blockchain::StorageValueTypes::storage_value_number:
371  return value.number_value == other.value.number_value;
372  case hsrcore::blockchain::StorageValueTypes::storage_value_bool:
373  return value.bool_value == other.value.bool_value;
374  return value.userdata_value == other.value.userdata_value;
375  case hsrcore::blockchain::StorageValueTypes::storage_value_null:
376  return true;
377  default: {
378  if (hsrcore::blockchain::is_any_table_storage_value_type(type)
379  || hsrcore::blockchain::is_any_array_storage_value_type(type))
380  {
381  if (value.table_value->size() != other.value.table_value->size())
382  return false;
383 
384  for (const auto &p1 : *(value.table_value))
385  {
386  auto found = other.value.table_value->find(p1.first);
387  if (found == other.value.table_value->end())
388  return false;
389  auto this_item = p1.second;
390  auto other_item = found->second;
391  if (!this_item.equals(other_item))
392  return false;
393  }
394  return true;
395  }
396  return false;
397  }
