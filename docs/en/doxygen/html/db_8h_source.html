124  template<typename K, typename T>
125  bool Read(const K& key, T& value)
126  {
127  if (!pdb)
128  return false;
129 
130  // Key
131  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
132  ssKey.reserve(1000);
133  ssKey << key;
134  Dbt datKey(&ssKey[0], ssKey.size());
135 
136  // Read
137  Dbt datValue;
138  datValue.set_flags(DB_DBT_MALLOC);
139  int ret = pdb->get(activeTxn, &datKey, &datValue, 0);
140  memset(datKey.get_data(), 0, datKey.get_size());
141  if (datValue.get_data() == NULL)
142  return false;
143 
144  // Unserialize value
145  try {
146  CDataStream ssValue((char*)datValue.get_data(), (char*)datValue.get_data() + datValue.get_size(), SER_DISK, CLIENT_VERSION);
147  ssValue >> value;
148  }
149  catch (std::exception &e) {
150  return false;
151  }
152 
153  // Clear and free memory
154  memset(datValue.get_data(), 0, datValue.get_size());
155  free(datValue.get_data());
156  return (ret == 0);

241  int ReadAtCursor(Dbc* pcursor, CDataStream& ssKey, CDataStream& ssValue, unsigned int fFlags = DB_NEXT)
242  {
243  // Read at cursor
244  Dbt datKey;
245  if (fFlags == DB_SET || fFlags == DB_SET_RANGE || fFlags == DB_GET_BOTH || fFlags == DB_GET_BOTH_RANGE)
246  {
247  datKey.set_data(&ssKey[0]);
248  datKey.set_size(ssKey.size());
249  }
250  Dbt datValue;
251  if (fFlags == DB_GET_BOTH || fFlags == DB_GET_BOTH_RANGE)
252  {
253  datValue.set_data(&ssValue[0]);
254  datValue.set_size(ssValue.size());
255  }
256  datKey.set_flags(DB_DBT_MALLOC);
257  datValue.set_flags(DB_DBT_MALLOC);
258  int ret = pcursor->get(&datKey, &datValue, fFlags);
259  if (ret != 0)
260  return ret;
261  else if (datKey.get_data() == NULL || datValue.get_data() == NULL)
262  return 99999;
263 
264  // Convert to streams
265  ssKey.SetType(SER_DISK);
266  ssKey.clear();
267  ssKey.write((char*)datKey.get_data(), datKey.get_size());
268  ssValue.SetType(SER_DISK);
269  ssValue.clear();
270  ssValue.write((char*)datValue.get_data(), datValue.get_size());
271 
272  // Clear and free memory
273  memset(datKey.get_data(), 0, datKey.get_size());
274  memset(datValue.get_data(), 0, datValue.get_size());
275  free(datKey.get_data());
